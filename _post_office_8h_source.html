293  void swapOutBoxes(Sender s)
294  {
295  //*********************************************************************
296  // If this sender is not registered, then there is nothing to do
297  //*********************************************************************
298  typename OutboxMap::iterator curOutPos = curOutboxes_i.find(s);
299  if(curOutboxes_i.end()==curOutPos)
300  {
301  return;
302  }
303 
304  //*********************************************************************
305  // Otherwise swap the list of outboxes for this sender in one go.
306  //*********************************************************************
307  prevOutboxes_i[s].swap(curOutPos->second);

311  //*********************************************************************

313  curIt!=curInboxes_i.end(); ++curIt)

317  // to the next receiver.

359  void addEdge(Sender s, Receiver r, const Message& msgVal)
360  {
361  //*********************************************************************
362  // If we have a message already, then we're done.
363  //*********************************************************************
364  PrivOutMsgMap& curOutMsgs = curOutboxes_i[s];
365  Message*& pCurOutMsg = curOutMsgs[r];
366  if(0!=pCurOutMsg)

368  return;
369  }
370 
371  //*********************************************************************
372  // Retrieve all other message pointers for this sender-receiver pair.
373  //*********************************************************************
374  PrivInMsgMap& curInMsgs = curInboxes_i[r];
375  PrivInMsgMap& prevInMsgs = prevInboxes_i[r];
376  PrivOutMsgMap& prevOutMsgs = prevOutboxes_i[s];

379  Message*& pPrevInMsg = prevInMsgs[s];

384  // pair are consistent.

386  assert(pCurInMsg==pCurOutMsg);
387  assert(pPrevInMsg==pPrevOutMsg);

389  assert( (0==pCurOutMsg) == (pCurOutMsg==pPrevOutMsg) );
