00851    template<typename F> void marginal
00852    (
00853     const DiscreteFunction& inFun,
00854     F aggregate,
00855     DiscreteFunction& outFun
00856    )
00857    {
00858       //************************************************************************
00859       // Ensure that the domain of outFun is a subset of inFun
00860       // We throw an exception if this is not the case.
00861       //************************************************************************
00862       if(!std::includes(inFun.varBegin(),inFun.varEnd(),
00863                outFun.varBegin(),outFun.varEnd()))
00864       {
00865          throw BadDomainException("marginal(DiscreteFunction,DiscreteFunction)",
00866                "Out domain is not subset of in domain.");
00867       }
00868 
00869       //************************************************************************
00870       // Calculate and store the result for each value in the output function.
00871       // Notice, we need to get the first element outside the loop, so that we
00872       // have at least two values with which to call the aggregate function.
00873       // If there is only one value, then that becomes its own aggregate.
00874       //************************************************************************
00875       for(DomainIterator outIt(outFun); outIt.hasNext(); ++outIt)
00876       {
00877          //*********************************************************************
00878          // Initialise the result to the first corresponding value in the input
00879          // function. This ensures that things like max and min will work
00880          // properly as aggregate functions, because they already have
00881          // something to work with.
00882          //*********************************************************************
00883          DomainIterator inIt(inFun);
00884          inIt.condition(outIt);
00885          ValType result = inFun(inIt.getInd());
00886          ++inIt;
00887 
00888          //*********************************************************************
00889          // Update the result by aggregating with any remaining values in the
00890          // free part of the input function domain.
00891          //*********************************************************************
00892          while(inIt.hasNext())
00893          {
00894             result = aggregate(result,inFun(inIt.getInd()));
00895             ++inIt;
00896          }
00897 
00898          //*********************************************************************
00899          // Store the current aggregate value in the current position of the
00900          // output function.
00901          //*********************************************************************
00902          outFun(outIt.getInd()) = result;
00903 
00904       } // for loop
