410 
411  assert(0==indices.size()); // sanity check for expected behaviour

413  //*********************************************************************
414  // Retrieve the indices that this function actually depends on.
415  // Notice this code assumes that all variable lists are sorted.
416  //*********************************************************************
417  VarIt v = varBegin;
418  IndIt i = indBegin;
419  std::vector<VarID>::const_iterator myV = vars_i.begin();
420  while( (i != indEnd) && (v != varEnd) && (myV != vars_i.end()) )
421  {
422  //******************************************************************
423  // If input variable is in the domain for this function...
424  //******************************************************************

557  {
558  //*********************************************************************
559  // Construct the union of the specified variables with this
560  // functions current domain
561  //*********************************************************************
562  std::vector<VarID> newVar = vars_i;
563  int maxSize = (end - begin) + vars_i.size();
564  newVar.reserve(maxSize);
565  newVar.insert(newVar.end(),begin,end);
566  std::sort(newVar.begin(),newVar.end());
567

569  std::unique(newVar.begin(),newVar.end());
570 
571  newVar.resize(newEnd-newVar.begin());
572

574  // If the specified list is a subset of the current domain, then
575  // we're done.
576  //*********************************************************************
577  if(newVar.size() <= vars_i.size())

635  template<class VarIt, class IndIt> void condition
636  (
637  VarIt vBegin,
638  VarIt vEnd,
639  IndIt iBegin,
640  IndIt iEnd
641  )
642  {
643  //*********************************************************************
644  // Construct an iterator for this function's domain, and condition
645  // on the specified variables.
646  //*********************************************************************
647  DomainIterator it(*this);
648  it.condition(vBegin,vEnd,iBegin,iEnd);
649 
650  //*********************************************************************
651  // If there are no variables to condition on (i.e. the intersection
652  // of the input variables with this variables domain is empty) then
653  // there is nothing left to do.
654  //*********************************************************************
655  if(0==it.fixedCount())
656  {
657  return;
658  }
659 
660  //*********************************************************************
661  // Otherwise construct the reduced domain of free variables.
662  //*********************************************************************
663  std::vector<VarID> freeVars;
664  freeVars.reserve(vars_i.size());
665  for(VarIterator varIt=varBegin(); varIt != varEnd(); ++varIt)
666  {
667  if(!it.isFixed(*varIt))
668  {
669  freeVars.push_back(*varIt);
670  }
671  }
672

674  // Create a temporary function to hold the result, and copy in the
675  // conditioned values.
676  //*********************************************************************
677  DiscreteFunction result(freeVars.begin(),freeVars.end());
678  while(it.hasNext())
679  {
680  result(it.getVars(),it.getSubInd()) = this->at(it.getInd());
681  ++it;
682  }
683

685  // Finally, swap the result values into this function
686  //*********************************************************************

851  template<typename F> void marginal
852  (
853  const DiscreteFunction& inFun,
854  F aggregate,
855  DiscreteFunction& outFun
856  )
857  {
858  //************************************************************************
859  // Ensure that the domain of outFun is a subset of inFun
860  // We throw an exception if this is not the case.
861  //************************************************************************
862  if(!std::includes(inFun.varBegin(),inFun.varEnd(),
863  outFun.varBegin(),outFun.varEnd()))
864  {
865  throw BadDomainException("marginal(DiscreteFunction,DiscreteFunction)",
866  "Out domain is not subset of in domain.");
867  }
868 
869  //************************************************************************
870  // Calculate and store the result for each value in the output function.
871  // Notice, we need to get the first element outside the loop, so that we
872  // have at least two values with which to call the aggregate function.
873  // If there is only one value, then that becomes its own aggregate.
874  //************************************************************************
875  for(DomainIterator outIt(outFun); outIt.hasNext(); ++outIt)
876  {
877  //*********************************************************************
878  // Initialise the result to the first corresponding value in the input
879  // function. This ensures that things like max and min will work
880  // properly as aggregate functions, because they already have
881  // something to work with.
882  //*********************************************************************
883  DomainIterator inIt(inFun);
884  inIt.condition(outIt);
885  ValType result = inFun(inIt.getInd());
886  ++inIt;
887 
888  //*********************************************************************
889  // Update the result by aggregating with any remaining values in the
890  // free part of the input function domain.
891  //*********************************************************************
892  while(inIt.hasNext())
893  {
894  result = aggregate(result,inFun(inIt.getInd()));
895  ++inIt;
896  }
897 
898  //*********************************************************************
899  // Store the current aggregate value in the current position of the
900  // output function.
901  //*********************************************************************
902  outFun(outIt.getInd()) = result;
903 
904  } // for loop
905 
906  } // function marginal
