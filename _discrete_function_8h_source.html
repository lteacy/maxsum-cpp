589  {

591  // Construct the union of the specified variables with this

593  //*********************************************************************

595  int maxSize = (end - begin) + vars_i.size();
596  newVar.reserve(maxSize);

607  // we're done.

931  template<typename F> void marginal
932  (
933  const DiscreteFunction& inFun,
934  F aggregate,
935  DiscreteFunction& outFun
936  )
937  {
938  //************************************************************************
939  // Ensure that the domain of outFun is a subset of inFun
940  // We throw an exception if this is not the case.
941  //************************************************************************
942  if(!std::includes(inFun.varBegin(),inFun.varEnd(),
943  outFun.varBegin(),outFun.varEnd()))
944  {
945  throw BadDomainException("marginal(DiscreteFunction,DiscreteFunction)",
946  "Out domain is not subset of in domain.");
947  }
948 
949  //************************************************************************
950  // Calculate and store the result for each value in the output function.
951  // Notice, we need to get the first element outside the loop, so that we
952  // have at least two values with which to call the aggregate function.
953  // If there is only one value, then that becomes its own aggregate.
954  //************************************************************************
955  for(DomainIterator outIt(outFun); outIt.hasNext(); ++outIt)
956  {
957  //*********************************************************************
958  // Initialise the result to the first corresponding value in the input
959  // function. This ensures that things like max and min will work
960  // properly as aggregate functions, because they already have
961  // something to work with.
962  //*********************************************************************
963  DomainIterator inIt(inFun);
964  inIt.condition(outIt);
965  ValType result = inFun(inIt.getInd());
966  ++inIt;
967 
968  //*********************************************************************
969  // Update the result by aggregating with any remaining values in the
970  // free part of the input function domain.
971  //*********************************************************************
972  while(inIt.hasNext())
973  {
974  result = aggregate(result,inFun(inIt.getInd()));
975  ++inIt;
976  }
977 
978  //*********************************************************************
979  // Store the current aggregate value in the current position of the
980  // output function.
981  //*********************************************************************
982  outFun(outIt.getInd()) = result;
983 
984  } // for loop
