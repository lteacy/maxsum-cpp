855  template<typename F> void marginal
856  (
857  const DiscreteFunction& inFun,
858  F aggregate,
859  DiscreteFunction& outFun
860  )
861  {
862  //************************************************************************
863  // Ensure that the domain of outFun is a subset of inFun
864  // We throw an exception if this is not the case.
865  //************************************************************************
866  if(!std::includes(inFun.varBegin(),inFun.varEnd(),
867  outFun.varBegin(),outFun.varEnd()))
868  {
869  throw BadDomainException("marginal(DiscreteFunction,DiscreteFunction)",
870  "Out domain is not subset of in domain.");
871  }
872 
873  //************************************************************************
874  // Calculate and store the result for each value in the output function.
875  // Notice, we need to get the first element outside the loop, so that we
876  // have at least two values with which to call the aggregate function.
877  // If there is only one value, then that becomes its own aggregate.
878  //************************************************************************
879  for(DomainIterator outIt(outFun); outIt.hasNext(); ++outIt)
880  {
881  //*********************************************************************
882  // Initialise the result to the first corresponding value in the input
883  // function. This ensures that things like max and min will work
884  // properly as aggregate functions, because they already have
885  // something to work with.
886  //*********************************************************************
887  DomainIterator inIt(inFun);
888  inIt.condition(outIt);
889  ValType result = inFun(inIt.getInd());
890  ++inIt;
891 
892  //*********************************************************************
893  // Update the result by aggregating with any remaining values in the
894  // free part of the input function domain.
895  //*********************************************************************
896  while(inIt.hasNext())
897  {
898  result = aggregate(result,inFun(inIt.getInd()));
899  ++inIt;
900  }
901 
902  //*********************************************************************
903  // Store the current aggregate value in the current position of the
904  // output function.
905  //*********************************************************************
906  outFun(outIt.getInd()) = result;
907 
908  } // for loop
