1144  (
1145  const DiscreteFunction& inFun,
1146  F aggregate,
1147  DiscreteFunction& outFun
1148  )
1149  {
1150  //************************************************************************
1151  // Ensure that the domain of outFun is a subset of inFun
1152  // We throw an exception if this is not the case.
1153  //************************************************************************
1154  if(!std::includes(inFun.varBegin(),inFun.varEnd(),
1155  outFun.varBegin(),outFun.varEnd()))
1156  {
1157  throw BadDomainException("marginal(DiscreteFunction,DiscreteFunction)",
1158  "Out domain is not subset of in domain.");
1159  }
1160 
1161  //************************************************************************
1162  // Calculate and store the result for each value in the output function.
1163  // Notice, we need to get the first element outside the loop, so that we
1164  // have at least two values with which to call the aggregate function.
1165  // If there is only one value, then that becomes its own aggregate.
1166  //************************************************************************
1167  for(DomainIterator outIt(outFun); outIt.hasNext(); ++outIt)
1168  {
1169  //*********************************************************************
1170  // Initialise the result to the first corresponding value in the input
1171  // function. This ensures that things like max and min will work
1172  // properly as aggregate functions, because they already have
1173  // something to work with.
1174  //*********************************************************************
1175  DomainIterator inIt(inFun);
1176  inIt.condition(outIt);
1177  ValType result = inFun(inIt.getInd());
1178  ++inIt;
1179 
1180  //*********************************************************************
1181  // Update the result by aggregating with any remaining values in the
1182  // free part of the input function domain.
1183  //*********************************************************************
1184  while(inIt.hasNext())
1185  {
1186  result = aggregate(result,inFun(inIt.getInd()));
1187  ++inIt;
1188  }
1189 
1190  //*********************************************************************
1191  // Store the current aggregate value in the current position of the
1192  // output function.
1193  //*********************************************************************
1194  outFun(outIt.getInd()) = result;
1195
