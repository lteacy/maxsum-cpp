94  //************************************************************************
95  if(0==ndims)

106  for(int k=0; k<ndims; k++)

114  //************************************************************************
115  int maxInd = cumProd[ndims-1] * siz[ndims-1] - 1;
116  if( (ind<0) || (ind > maxInd) )
117  {
118  throw OutOfRangeException("bad linear index","maxsum::ind2sub");
119  }
120 
121  //************************************************************************

123  // the cumulative dimension sizes, and recording the remainder.
124  // Have to start at the largest dimension and work backwards.
125  //************************************************************************

127  for(int k=ndims-1; k>=0; k--)
